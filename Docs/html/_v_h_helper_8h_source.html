93  static std::array<VkVertexInputAttributeDescription, 4> getAttributeDescriptions() {
94  std::array<VkVertexInputAttributeDescription, 4> attributeDescriptions = {};
95 
96  attributeDescriptions[0].binding = 0;
97  attributeDescriptions[0].location = 0;
98  attributeDescriptions[0].format = VK_FORMAT_R32G32B32_SFLOAT;
99  attributeDescriptions[0].offset = offsetof(vhVertex, pos);
100 
101  attributeDescriptions[1].binding = 0;
102  attributeDescriptions[1].location = 1;
103  attributeDescriptions[1].format = VK_FORMAT_R32G32B32_SFLOAT;
104  attributeDescriptions[1].offset = offsetof(vhVertex, normal);
105 
106  attributeDescriptions[2].binding = 0;
107  attributeDescriptions[2].location = 2;
108  attributeDescriptions[2].format = VK_FORMAT_R32G32B32_SFLOAT;
109  attributeDescriptions[2].offset = offsetof(vhVertex, tangent);
110 
111  attributeDescriptions[3].binding = 0;
112  attributeDescriptions[3].location = 3;
113  attributeDescriptions[3].format = VK_FORMAT_R32G32_SFLOAT;
114  attributeDescriptions[3].offset = offsetof(vhVertex, texCoord);
115 
116  return attributeDescriptions;
117  }
