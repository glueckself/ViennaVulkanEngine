97  static std::array<VkVertexInputAttributeDescription, 4> getAttributeDescriptions() {
98  std::array<VkVertexInputAttributeDescription, 4> attributeDescriptions = {};
99 
100  attributeDescriptions[0].binding = 0;
101  attributeDescriptions[0].location = 0;
102  attributeDescriptions[0].format = VK_FORMAT_R32G32B32_SFLOAT;
103  attributeDescriptions[0].offset = offsetof(vhVertex, pos);
104 
105  attributeDescriptions[1].binding = 0;
106  attributeDescriptions[1].location = 1;
107  attributeDescriptions[1].format = VK_FORMAT_R32G32B32_SFLOAT;
108  attributeDescriptions[1].offset = offsetof(vhVertex, normal);
109 
110  attributeDescriptions[2].binding = 0;
111  attributeDescriptions[2].location = 2;
112  attributeDescriptions[2].format = VK_FORMAT_R32G32B32_SFLOAT;
113  attributeDescriptions[2].offset = offsetof(vhVertex, tangent);
114 
115  attributeDescriptions[3].binding = 0;
116  attributeDescriptions[3].location = 3;
117  attributeDescriptions[3].format = VK_FORMAT_R32G32_SFLOAT;
118  attributeDescriptions[3].offset = offsetof(vhVertex, texCoord);
119 
120  return attributeDescriptions;
121  }
122 
124  bool operator==(const vhVertex& other) const {
125  return pos == other.pos && normal == other.normal && tangent == other.tangent && texCoord == other.texCoord;
126  }
